768 static void
769 addEdgeConstraint( Polytope& P, unsigned int i, unsigned int j,
770 const std::vector<Point>& pts )
771 {
772 Vector ab = pts[ i ] - pts[ j ];
773 for ( int s = 0; s < 2; s++ )
774 for ( Dimension k = 0; k < dimension; ++k )
775 {
776 Vector n = ab.crossProduct( Point::base( k, (s == 0) ? 1 : -1 ) );
777 Integer b = n.dot( pts[ i ] );
778 std::size_t nb_in = 0;
779 for ( auto p : pts ) {
780 Integer v = n.dot( p );
781 if ( v < b ) nb_in++;
782 }
783 if ( nb_in == dimension - 1 ) {
784 P.cut( n, b, true, true );
785 }
786 }
787 }
792 static
793 Vector crossProduct( const Vector& v1, const Vector& v2 )
794 {
795 return v1.crossProduct( v2 );
796 }
797 };
