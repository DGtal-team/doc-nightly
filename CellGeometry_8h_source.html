458 getIncidentCellsToPointels( const KSpace& K,
459 PointelIterator itB, PointelIterator itE )
460 {
461 std::unordered_set<typename KSpace::Cell> cells;
462 if ( i == 0 )
463 for ( auto it = itB; it != itE; ++it )

465 else
466 for ( auto it = itB; it != itE; ++it )

468 auto pointel = *it;
